56def _parse_headers_and_cookies( 

57 parsed_args: argparse.Namespace, 

58) -> tuple[list[tuple[str, bytes]], dict[str, str]]: 

59 headers: list[tuple[str, bytes]] = [] 

60 cookies: dict[str, str] = {} 

61 for header in parsed_args.headers or (): 

62 name, val = header.split(":", 1) 

63 name = name.strip() 

64 val = val.strip() 

65 if name.title() == "Cookie": 

66 for name, morsel in SimpleCookie(val).items():

78def curl_to_request_kwargs(

80) -> dict[str, Any]: 

81 """Convert a cURL command syntax to Request kwargs. 

82 

83 :param str curl_command: string containing the curl command 

84 :param bool ignore_unknown_options: If true, only a warning is emitted when 

85 cURL options are unknown. Otherwise 

86 raises an error. (default: True) 

87 :return: dictionary of Request kwargs 

88 """

93 raise ValueError('A curl command must start with "curl"')

95 parsed_args, argv = curl_parser.parse_known_args(curl_args[1:]) 

96 

97 if argv: 

98 msg = f"Unrecognized options: {', '.join(argv)}" 

99 if ignore_unknown_options: 

100 warnings.warn(msg) 

101 else: 

102 raise ValueError(msg) 

103 

104 url = parsed_args.url 

105 

106 # curl automatically prepends 'http' if the scheme is missing, but Request 

107 # needs the scheme to work 

108 parsed_url = urlparse(url) 

109 if not parsed_url.scheme: 

110 url = "http://" + url 

111 

112 method = parsed_args.method or "GET" 

113 

114 result: dict[str, Any] = {"method": method.upper(), "url": url} 

115 

116 headers, cookies = _parse_headers_and_cookies(parsed_args) 

117 

118 if headers: 

119 result["headers"] = headers 

120 if cookies: 

121 result["cookies"] = cookies 

122 if parsed_args.data: 

123 result["body"] = parsed_args.data

125 # if the "data" is specified but the "method" is not specified, 

126 # the default method is 'POST'
