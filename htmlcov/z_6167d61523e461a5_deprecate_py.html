33 """ 

34 Return a "deprecated" class that causes its subclasses to issue a warning. 

35 Subclasses of ``new_class`` are considered subclasses of this class. 

36 It also warns when the deprecated class is instantiated, but do not when 

37 its subclasses are instantiated. 

38 

39 It can be used to rename a base class in a library. For example, if we 

40 have 

41 

42 class OldName(SomeClass): 

43 # ... 

44 

45 and we want to rename it to NewName, we can do the following:: 

46 

47 class NewName(SomeClass): 

48 # ... 

49 

50 OldName = create_deprecated_class('OldName', NewName) 

51 

52 Then, if user class inherits from OldName, warning is issued. Also, if 

53 some code uses ``issubclass(sub, OldName)`` or ``isinstance(sub(), OldName)`` 

54 checks they'll still return True if sub is a subclass of NewName instead of 

55 OldName. 

56 """

164def method_is_overridden(subclass: type, base_class: type, method_name: str) -> bool: 

165 """ 

166 Return True if a method named ``method_name`` of a ``base_class`` 

167 is overridden in a ``subclass``. 

168 

169 >>> class Base: 

170 ... def foo(self): 

171 ... pass 

172 >>> class Sub1(Base): 

173 ... pass 

174 >>> class Sub2(Base): 

175 ... def foo(self): 

176 ... pass 

177 >>> class Sub3(Sub1): 

178 ... def foo(self): 

179 ... pass 

180 >>> class Sub4(Sub2): 

181 ... pass 

182 >>> method_is_overridden(Sub1, Base, 'foo') 

183 False 

184 >>> method_is_overridden(Sub2, Base, 'foo') 

185 True 

186 >>> method_is_overridden(Sub3, Base, 'foo') 

187 True 

188 >>> method_is_overridden(Sub4, Base, 'foo') 

189 True 

190 """ 

191 base_method = getattr(base_class, method_name) 

192 sub_method = getattr(subclass, method_name)
